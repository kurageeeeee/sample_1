    const TARGET_URL = "https://youtu.be/awA9gPL94jY?si=uK7L4ckAkoqUFfKM";

    document.body.addEventListener('click', startPrank, { once: true });

    function startPrank() {

        if (document.documentElement.requestFullscreen) {

            document.documentElement.requestFullscreen().catch(() => {});

        }

        statusText.innerHTML = "ウイルスを検出しました。<br>システム保護のため全データを削除します...";

        statusText.style.color = "red";

       

        pContainer.style.display = "block";

        pText.style.display = "block";

        overlay.style.display = "block";

        let width = 0;

       

        const interval = setInterval(() => {

            width += Math.random() * 2;

           

            if (width >= 100) {

                width = 100;

                clearInterval(interval);

                showResult();

            }

            pBar.style.width = width + "%";

            pText.innerText = Math.floor(width) + "% DELETED";

        }, 50);

    }

    function showResult() {

        setTimeout(() => {

            document.body.style.backgroundColor = "#fff";

            setTimeout(() => {

                document.body.style.backgroundColor = "#333";

                warningBox.style.display = "none";

                pContainer.style.display = "none";

                pText.style.display = "none";

                overlay.style.display = "none";

               

                // ネタばらし表示

                peace.style.display = "block";

               

                if (document.exitFullscreen) {

                    document.exitFullscreen().catch(() => {});

                }

                // ★ ここでYouTubeに飛ばす処理 ★

                // 1500ミリ秒（1.5秒）後にジャンプ！

                    window.location.href = TARGET_URL;

                }, 1500);

            }, 100);

        }, 500);

    }
